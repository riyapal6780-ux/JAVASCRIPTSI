<script>
function calculateResult() {
    // 1. Fetch values from input boxes and convert strings to numbers
    let mathMarks = parseFloat(document.getElementById('math').value);
    let scienceMarks = parseFloat(document.getElementById('science').value);
    let englishMarks = parseFloat(document.getElementById('english').value);
    let historyMarks = parseFloat(document.getElementById('history').value);

    // 2. Validate inputs to verify fields are not left blank
    if (isNaN(mathMarks) || isNaN(scienceMarks) || isNaN(englishMarks) || isNaN(historyMarks)) {
        document.getElementById('resultOutput').innerHTML = "<span style='color: red;'>Please enter marks for all fields!</span>";
        return;
    }

    // 3. Compute basic math statistics 
    let totalMarks = mathMarks + scienceMarks + englishMarks + historyMarks;
    let percentage = (totalMarks / 400) * 100;
    
    // 4. Use conditional statements to determine pass/fail status and final grade
    let passOrFail = "";
    if (mathMarks >= 40 && scienceMarks >= 40 && englishMarks >= 40 && historyMarks >= 40) {
        passOrFail = "<span style='color: green; font-weight: bold;'>Passed</span>";
    } else {
        passOrFail = "<span style='color: red; font-weight: bold;'>Failed (Must score 40+ in all subjects)</span>";
    }

    let grade = "";
    if (percentage >= 90) {
        grade = "A+";
    } else if (percentage >= 80) {
        grade = "A";
    } else if (percentage >= 70) {
        grade = "B";
    } else if (percentage >= 60) {
        grade = "C";
    } else if (percentage >= 40) {
        grade = "D";
    } else {
        grade = "F";
    }

    // 5. Update HTML layout dynamically with calculations
    document.getElementById('resultOutput').innerHTML = `
        <hr>
        <strong>Total Marks:</strong> ${totalMarks} / 400 <br>
        <strong>Percentage:</strong> ${percentage.toFixed(2)}% <br>
        <strong>Grade:</strong> ${grade} <br>
        <strong>Status:</strong> ${passOrFail}
    `;
}
</script>
