    <br><br>

    <h3>Output:</h3>
    <p id="v"></p>

    <script>
        $(document).ready(function () {

            $("#txt").keyup(function () {

                // Alert on every key press
                alert("Key Pressed");

                // Get input value
                let n = $("#txt").val();

                // Display value
                $("#v").html(n);

            });

        });
    </script>
